47export async function register(data: RegisterData): Promise<void> {
48 try {
49 const response = await fetch(`${API_BASE_URL}/user/register`, {
50 method: "POST",
51 headers: {
52 "Content-Type": "application/json",
53 },
54 body: JSON.stringify(data),
55 });
56
57 if (!response.ok) {
58 const errorData = (await response.json()) as AuthError;
59 throw new Error(errorData.detail || "Registration failed");
60 }
61
62 const authData = (await response.json()) as AuthResponse;
63 if (authData.access_token) {
64 localStorage.setItem(TOKEN_KEY, authData.access_token);
65 }
66 } catch (error) {
67 if (error instanceof Error) {
68 throw error;
69 }
70 throw new Error("Network error during registration");
71 }
72}
