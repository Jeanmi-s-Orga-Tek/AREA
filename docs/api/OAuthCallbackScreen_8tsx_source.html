8import React, { useEffect, useState } from "react";
9import { handleOAuthCallback } from "../services/auth";
10
11const OAuthCallbackScreen: React.FC = () => {
12 const [error, setError] = useState<string | null>(null);
13 const [loading, setLoading] = useState(true);
14
15 useEffect(() => {
16 const processOAuthCallback = async () => {
17 try {
18 if (window.location.hostname === "127.0.0.1") {
19 const newUrl = window.location.href.replace("127.0.0.1", "localhost");
20 window.location.href = newUrl;
21 return;
22 }
23
24 const urlParams = new URLSearchParams(window.location.search);
25
26 const hash = window.location.hash;
27 const hashContent = hash.replace(/^#/, '');
28
29 let hashToken = null;
30 let hashState = null;
31 if (hashContent) {
32 const hashPairs = hashContent.split('&');
33 for (const pair of hashPairs) {
34 const [key, value] = pair.split('=');
35 if (key === 'token') hashToken = decodeURIComponent(value);
36 if (key === 'state') hashState = decodeURIComponent(value);
37 }
38 }
39
40 let code = urlParams.get("code") || urlParams.get("token") || hashToken;
41 let state = urlParams.get("state") || hashState;
42 const errorParam = urlParams.get("error");
43
44 if (errorParam) {
45 const errorDescription = urlParams.get("error_description") || errorParam;
46 throw new Error(`Erreur OAuth: ${errorDescription}`);
47 }
48
49 if (!code) {
50 throw new Error("Code manquant dans l'URL de callback");
51 }
52
53 const stateValue = state || "";
54
55 const pathParts = window.location.pathname.split("/");
56 const provider = pathParts[pathParts.length - 1];
57
58 if (!provider) {
59 throw new Error("Provider non identifié dans l'URL");
60 }
61
62 await handleOAuthCallback(provider, code, stateValue);
63
64 window.location.href = "/";
65 } catch (err) {
66 console.error("OAuth callback error:", err);
67 setError(err instanceof Error ? err.message : "Une erreur est survenue");
68 setLoading(false);
69
70 setTimeout(() => {
71 window.location.href = "/login";
72 }, 3000);
73 }
74 };
75
76 processOAuthCallback();
77 }, []);
78
79 if (loading) {
80 return (
81 <div style={styles.container}>
82 <div style={styles.card}>
83 <div style={styles.spinner}></div>
84 <h2 style={styles.title}>Authentification en cours...</h2>
85 <p style={styles.message}>Veuillez patienter pendant que nous vous connectons.</p>
86 </div>
87 </div>
88 );
89 }
90
91 if (error) {
92 return (
93 <div style={styles.container}>
94 <div style={styles.card}>
95 {/* <div style={styles.errorIcon}></div> */}
96 <h2 style={styles.title}>Erreur d'authentification</h2>
97 <p style={styles.errorMessage}>{error}</p>
98 <p style={styles.message}>Vous allez être redirigé vers la page de connexion...</p>
99 </div>
100 </div>
101 );
102 }
103
104 return null;
105};
