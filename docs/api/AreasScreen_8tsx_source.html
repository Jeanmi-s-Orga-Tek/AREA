46 try {
47 setLoading(true);
48 setError("");
49
50 const apiAreas = await fetchMyAreas();
51
52 const mappedAreas: Area[] = apiAreas.map((area) => ({
53 id: area.id,
54 name: area.name,
55 action: {
56 service: area.action.service.display_name || area.action.service.name,
57 type: area.action.action.name,
58 description: area.action.action.description,
59 parameters: area.action_parameters || {},
60 },
61 reaction: {
62 service: area.reaction.service.display_name || area.reaction.service.name,
63 type: area.reaction.reaction.name,
64 description: area.reaction.reaction.description,
65 parameters: area.reaction_parameters || {},
66 },
67 isActive: area.is_active,
68 createdAt: new Date(area.created_at).toLocaleDateString("fr-FR", {
69 day: "2-digit",
70 month: "short",
71 year: "numeric",
72 }),
73 }));
74
75 setAreas(mappedAreas);
76 } catch (err) {
77 setError(err instanceof Error ? err.message : "Erreur lors du chargement des AREAs");

80 setLoading(false);

84 const handleToggleActive = async (id: number) => {

93 alert(`Erreur: ${err instanceof Error ? err.message : "Erreur inconnue"}`);

108 }
109 }

112 const activeCount = areas.filter((area) => area.isActive).length;

169 area.isActive
