105def subscribe_service(service_id: int, session: SessionDep, token: TokenDep):
106 user = get_user_from_token(token, session)
107 if user is None or user.id is None:
108 raise HTTPException(status_code=status.HTTP_400_BAD_REQUEST, detail="Invalid user")
109 service = session.get(Service, service_id)

114 UserServiceSubscription.service_id == service_id,
115 )

119 session.add(subscription)
