8import React, { useState, useEffect } from "react";
9import { register, fetchOAuthProviders, initiateOAuthLogin, OAuthProvider } from "../services/auth";
10import "./RegisterScreen.css";
11
12const RegisterScreen: React.FC = () => {
13 const [email, setEmail] = useState("");
14 const [username, setUsername] = useState("");
15 const [password, setPassword] = useState("");
16 const [confirmPassword, setConfirmPassword] = useState("");
17 const [error, setError] = useState("");
18 const [success, setSuccess] = useState(false);
19 const [loading, setLoading] = useState(false);
20 const [oauthProviders, setOauthProviders] = useState<OAuthProvider[]>([]);
21 const [loadingProviders, setLoadingProviders] = useState(true);
22
23 useEffect(() => {
24 const loadProviders = async () => {
25 try {
26 const providers = await fetchOAuthProviders();
27 setOauthProviders(providers.filter(p => p.available && p.flows.web));
28 } catch (err) {
29 console.error("Failed to load OAuth providers:", err);
30 } finally {
31 setLoadingProviders(false);
32 }
33 };
34
35 loadProviders();
36 }, []);
37
38 const demoProviders = ["Google", "GitHub", "Discord", "Spotify", "Microsoft", "Trello"];
39
40 const handleSubmit = async (e: React.FormEvent) => {
41 e.preventDefault();
42 setError("");
43 setSuccess(false);
44
45 if (!email || !username || !password || !confirmPassword) {
46 setError("Tous les champs sont requis");
47 return;
48 }
49
50 if (password !== confirmPassword) {
51 setError("Les mots de passe ne correspondent pas");
52 return;
53 }
54
55 if (password.length < 6) {
56 setError("Le mot de passe doit contenir au moins 6 caractères");
57 return;
58 }
59
60 setLoading(true);
61
62 try {
63 await register({ email, name: username, new_password: password });
64 setSuccess(true);
65 setEmail("");
66 setUsername("");
67 setPassword("");
68 setConfirmPassword("");
69 setTimeout(() => {
70 window.location.href = "/login";
71 }, 2000);
72 } catch (err) {
73 setError(err instanceof Error ? err.message : "Une erreur est survenue");
74 } finally {
75 setLoading(false);
76 }
77 };
78
79 const handleOAuthLogin = async (providerId: string) => {
80 setError("");
81 setLoading(true);
82
83 try {
84 await initiateOAuthLogin(providerId);
85 } catch (err) {
86 setError(err instanceof Error ? err.message : `Erreur lors de l'inscription OAuth`);
87 setLoading(false);
88 }
89 };
90
91 return (
92 <div className="register-container">
93 <div className="register-form-card">
94 <h1 className="register-title">Inscription</h1>
95 <p className="register-subtitle">Créez votre compte AREA</p>
96
97 {error && <div className="register-error-box">{error}</div>}
98 {success && (
99 <div className="register-success-box">
100 Inscription réussie ! Redirection vers la page de connexion...
101 </div>
102 )}
103
104 <form onSubmit={handleSubmit} className="register-form">
105 <div className="register-form-group">
106 <label htmlFor="email" className="register-label">
107 Email
108 </label>
109 <input
110 type="email"
111 id="email"
112 value={email}
113 onChange={(e) => setEmail(e.target.value)}
114 className="register-input"
115 placeholder="votre@email.com"
116 disabled={loading}
117 />
118 </div>
119
120 <div className="register-form-group">
121 <label htmlFor="username" className="register-label">
122 Nom d'utilisateur
123 </label>
124 <input
125 type="text"
126 id="username"
127 value={username}
128 onChange={(e) => setUsername(e.target.value)}
129 className="register-input"
130 placeholder="John Doe"
131 disabled={loading}
132 />
133 </div>
134
135 <div className="register-form-group">
136 <label htmlFor="password" className="register-label">
137 Mot de passe
138 </label>
139 <input
140 type="password"
141 id="password"
142 value={password}
143 onChange={(e) => setPassword(e.target.value)}
144 className="register-input"
145 placeholder="••••••••"
146 disabled={loading}
147 />
148 </div>
149
150 <div className="register-form-group">
151 <label htmlFor="confirmPassword" className="register-label">
152 Confirmer le mot de passe
153 </label>
154 <input
155 type="password"
156 id="confirmPassword"
157 value={confirmPassword}
158 onChange={(e) => setConfirmPassword(e.target.value)}
159 className="register-input"
160 placeholder="••••••••"
161 disabled={loading}
162 />
163 </div>
164
165 <button
166 type="submit"
167 className="register-button"
168 disabled={loading}
169 >
170 {loading ? "Inscription en cours..." : "S'inscrire"}
171 </button>
172 </form>
173
174 <div className="register-divider">
175 <span>OU</span>
176 </div>
177
178 <div className="register-oauth-buttons">
179 {loadingProviders ? (
180 <p style={{ textAlign: "center", color: "#999" }}>Chargement des providers...</p>
181 ) : oauthProviders.length === 0 ? (
182 <p style={{ textAlign: "center", color: "#999" }}>Aucun provider OAuth2 disponible</p>
183 ) : (
184 oauthProviders.map((provider) => (
185 <button
186 key={provider.id}
187 onClick={() => handleOAuthLogin(provider.id)}
188 className="register-oauth-button"
189 disabled={loading}
190 style={{
191 borderLeft: `4px solid ${provider.color}`,
192 }}
193 >
194 <span style={{ marginRight: "8px" }}>{provider.icon}</span>
195 S'inscrire avec {provider.name}
196 </button>
197 ))
198 )}
199 </div>
200
201 <div className="register-footer">
202 <p className="register-footer-text">
203 Vous avez déjà un compte ?{" "}
204 <a href="/login" className="register-link">
205 Se connecter
206 </a>
207 </p>
208 </div>
209 </div>
210 </div>
211 );
212};

214export default RegisterScreen;
