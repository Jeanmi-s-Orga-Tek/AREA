148function getAuthHeaders(): Record<string, string> {
149 const token = getToken();
150 const headers: Record<string, string> = {
151 "Content-Type": "application/json",
152 Accept: "application/json",
153 };
154 if (token) {
155 headers.Authorization = `Bearer ${token}`;
156 }
157 return headers;
158}

160async function handleResponse<T>(response: Response): Promise<T> {
161 if (!response.ok) {
162 const errorData = await response.json().catch(() => ({ detail: "Request failed" }));
163 throw new Error(errorData.detail || `Request failed with status ${response.status}`);
164 }
165 return response.json();
166}

172export async function fetchAbout(): Promise<AboutResponse> {
173 try {
174 const response = await fetch(ABOUT_URL, {
175 headers: {
176 Accept: "application/json",
177 },
178 });
179
180 if (!response.ok) {
181 throw new Error(
182 `Failed to fetch /about.json: ${response.status} ${response.statusText}`
183 );
184 }
185
186 const data = (await response.json()) as AboutResponse;
187 return data;
188 } catch (error) {
189 const message =
190 error instanceof Error ? error.message : "Unknown network error";
191 throw new Error(`Unable to reach /about.json: ${message}`);
192 }
193}
