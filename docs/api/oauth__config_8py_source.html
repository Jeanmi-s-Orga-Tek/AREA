34def _expand_env_vars(value: str) -> str:
35 if isinstance(value, str):
36 pattern = r'\$\{([^}]+)\}'
37 matches = re.findall(pattern, value)
38 for var_name in matches:
39 env_value = os.environ.get(var_name, "")
40 if not env_value:
41 raise ValueError(f"Environment variable '{var_name}' is not set")
42 value = value.replace(f"${{{var_name}}}", env_value)
43 return value
